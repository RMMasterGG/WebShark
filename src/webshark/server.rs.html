35    /// Запускает HTTP-сервер на указанном сетевом адресе.
36    ///
37    /// Принимает адрес (например, `"127.0.0.1:8080"`) и настроенный [`Router`].
38    /// Оборачивает роутер в [`std::sync::Arc`] для безопасного совместного использования.
39    ///
40    /// # Errors
41    ///
42    /// Возвращает [`std::io::Error`], если не удалось забиндить указанный порт.
43    ///
44    /// # Examples
45    ///
46    /// ```no_run
47    /// use webshark::start_server;
48    /// use webshark::routing::request::{Method, Request};
49    /// use webshark::routing::response::Response;
50    /// use webshark::routing::route::Route;
51    /// use webshark::routing::router::Router;
52    ///
53    ///#[tokio::main]
54    /// async fn main() {
55    ///     // Создаём основной обработчик маршрутов
56    ///     let mut router = Router::new();
57    ///
58    ///     // Добавляем в обработчик маршрутов новый маршрут
59    ///     router.add_route(Route::new(Method::GET, "/", test_handler));
60    ///
61    ///     // Запускаем сервер
62    ///     start_server(router).await.unwrap();
63    /// }
64    ///
65    /// fn test_handler(request: Request) -> Response {
66    ///     let response = Response::ok();
67    ///     println!("{:#}", response);
68    ///     response
69    /// }
70    /// ```
71    pub async fn start_server(&self) -> std::io::Result<()> {
72        Config::init_config();
73        let config = APP_CONFIG.get().unwrap();
74
75        let format = fmt::format()
76            .with_thread_ids(true)
77            .with_thread_names(true)
78            .compact();
79
80        // Logging subscribe
81        tracing_subscriber::fmt().event_format(format).init();
82
83        let tcp_listener = TcpListener::bind(config.server().server_and_port()).await?;
84
85        let router = Arc::new(self.router.clone());
86
87        println!("{}", SHARK_BANNER);
88
89        println!("HTTP-сервер успешно запущен и слушает подключения...");
90
91        loop {
92            let (stream, _) = tcp_listener.accept().await?;
93            let router_clone = router.clone();
94
95            tokio::spawn(async move {
96                handle_client(stream, router_clone).await;
97            });
98        }
99    }
100}

102/// Проверяет входящий HTTP-запрос на соответствие политикам безопасности CORS и CSRF.
103///
104/// # Назначение
105/// Метод выполняет многоуровневую фильтрацию сетевого трафика:
106/// 1. Разрешает прямые не-браузерные запросы (например, от `curl`, Postman или ручного ввода URL).
107/// 2. Защищает от CSRF-атак (Cross-Site Request Forgery), валидируя заголовок `Referer` при отсутствии `Origin`.
108/// 3. Проверяет домен отправителя (`Origin`) на соответствие белому списку из конфигурации приложения.
109/// 4. Обрабатывает предварительные CORS-запросы (Preflight/`OPTIONS`), проверяя запрашиваемые методы и кастомные заголовки.
110///
111/// # Возвращаемое значение
112/// * `true` — запрос полностью валиден и безопасен, его можно передавать дальше по цепочке фильтров к роутеру.
113/// * `false` — обнаружено нарушение политик безопасности, запрос должен быть немедленно заблокирован с ответом `403 Forbidden` (или `400 Bad Request`).
114///
115/// # Литература и стандарты
116/// * [Семейство спецификаций W3C CORS](https://w3.org) [1]
117/// * [RFC 9110: HTTP Semantics (Раздел 12: Заголовки авторизации и происхождения)](https://ietf.org) [2]
118fn validate_cors_and_origin(request: &Request<Bytes>) -> bool {
119    // TODO: [Производительность] Избежать распаковки глобального конфига при каждом вызове.
120    // Передавать `&Config` в качестве аргумента функции или перенести метод внутрь структуры `FilterChain`.
121    let config = match APP_CONFIG.get() {
122        Some(c) => c,
123        None => return true,
124    };
125
126    let origin = request.get_header(ORIGIN).unwrap_or("");
127    let referer = request.get_header(REFERER).unwrap_or("");
128
129    // TODO: [Безопасность / Host Injection] Добавить обязательную сверку заголовка `Host` из запроса
130    // с реальным `config.server().server_and_port()`. Если они не совпадают — блокировать запрос,
131    // чтобы предотвратить подмену хоста (Host Header Attack).
132
133    // 1. Прямой запрос без заголовков происхождения (ввод URL руками в браузере или curl) — разрешаем
134    if origin.is_empty() && referer.is_empty() {
135        return true;
136    }
137
138    // 2. Защита от CSRF по Referer (если запрос инициирован переходом по ссылке со стороннего ресурса)
139    if origin.is_empty() && !referer.is_empty() {
140        let host = config.server().server_and_port();
141
142        // TODO: [Продакшен Оптимизация] Избежать аллокаций динамических строк `format!` через кучу при каждом запросе.
143        // Заранее сгенерировать строки "http://..." и "https://..." на этапе `init_config` и сохранить их в конфиг,
144        // либо проверять вхождение через побайтовое сравнение срезов.
145        let is_valid_http = referer.contains(format!("http://{}", host).as_str());
146        let is_valid_https = referer.contains(format!("https://{}", host).as_str());
147
148        // TODO: [Безопасность / Строгий парсинг] Метод `.contains` уязвим, если атакующий создаст домен вида `localhost:://evil.com`.
149        // Реферер должен парситься как полноценный URL (выделять схему и хост) и сравниваться на точное равенство с `host`.
150        if !is_valid_http && !is_valid_https {
151            return false;
152        }
153
154        return true;
155    }
156
157    // TODO: [Продакшен / Reverse Proxy] Если сервер стоит за Nginx, Cloudflare или балансировщиком нагрузки,
158    // заголовок `Origin` может содержать IP прокси. Нужно добавить чтение заголовков `X-Forwarded-Host` или `Forwarded`.
159
160    // 3. Проверка домена (CORS Origin)
161    if !config.cors().is_allowed_origin(origin) {
162        return false;
163    }
164
165    if config.cors().allowed_origins().contains(&"*".to_string()) {
166        return true;
167    }
168
169    // 4. Проверки для Preflight-запросов (OPTIONS)
170    if request.method() == Method::OPTIONS {
171        let request_method = request
172            .get_header(ACCESS_CONTROL_REQUEST_METHOD)
173            .unwrap_or("");
174
175        if !request_method.is_empty() {
176            if let Ok(req_method) = Method::from_bytes(request_method.as_bytes()) {
177                if !config.cors().is_allowed_method(req_method) {
178                    return false;
179                }
180            } else {
181                return false;
182            }
183        }
184
185        if let Some(req_headers_str) = request.get_header(ACCESS_CONTROL_REQUEST_HEADERS)
186            && !req_headers_str.is_empty()
187        {
188            let all_headers_allowed = req_headers_str
189                .split(",")
190                .all(|h| config.cors().is_allowed_header(h));
191
192            if !all_headers_allowed {
193                return false;
194            }
195        }
196
197        // TODO: [CORS Спецификация / Куки] Если в `config.toml` включена поддержка сессий/кук (`Allow-Credentials`),
198        // то по стандарту браузеров `allowed_origins` НЕ МОЖЕТ содержать маску `*`.
199        // Добавить проверку: если `origin == "*"` и `credentials == true`, возвращать `false` (защита от падения фронтенда).
200    }
201
202    true
203}

205/// Обрабатывает подключение конкретного TCP-клиента.
206///
207/// Метод парсит входящие байты в [`Request`], ищет подходящий эндпоинт
208/// в [`Router`], вызывает его и отправляет полученный [`Response`] обратно в сеть.
209async fn handle_client<T>(mut stream: T, routers: Arc<Router>)
210where
211    T: AsyncRead + AsyncWrite + Unpin + Send + 'static + Sync,
212{
213    let request = match Request::parse(&mut stream).await {
214        Ok(req) => req,
215        Err(e) => return error!("Error parsing request: {}", e),
216    };
217
218    if !validate_cors_and_origin(&request) {
219        let response = Response::forbidden();
220        let _ = response.send(&mut stream).await;
221        let _ = stream.flush().await;
222        return;
223    }
224
225    let client_origin = request.get_header(ORIGIN).map(|s| s.to_string());
226
227    if request.method() == Method::OPTIONS {
228        let config = APP_CONFIG.get().unwrap();
229
230        let origin_str = client_origin.as_deref().unwrap_or("*");
231
232        let allowed_methods_str = config
233            .cors()
234            .allowed_methods()
235            .iter()
236            .map(|method| method.as_str())
237            .collect::<Vec<&str>>()
238            .join(", ");
239
240        // Отправляем пустой ответ 204 с разрешениями
241        let cors_response = Response::no_content()
242            .header(ACCESS_CONTROL_ALLOW_ORIGIN, origin_str)
243            .header(ACCESS_CONTROL_ALLOW_METHODS, allowed_methods_str)
244            .header(
245                ACCESS_CONTROL_ALLOW_HEADERS,
246                config.cors().allowed_headers().join(", "),
247            );
248
249        if let Err(e) = cors_response.send(&mut stream).await {
250            println!("Ошибка отправки предзапроса OPTIONS: {}", e);
251        }
252        return;
253    }
254
255    let req_method = request.method().clone();
256    let req_path = request.uri().path().to_string();
257
258    let is_connection = request
259        .get_header(CONNECTION)
260        .map(|value| {
261            value
262                .split(",")
263                .any(|v| v.trim().eq_ignore_ascii_case("upgrade"))
264        })
265        .unwrap_or(false);
266
267    let is_upgrade = request
268        .get_header(UPGRADE)
269        .map(|value| value.eq_ignore_ascii_case("websocket"))
270        .unwrap_or(false);
271
272    let is_websocket = is_connection && is_upgrade;
273
274    let router_type = if is_websocket {
275        RouteType::WebSocket
276    } else {
277        RouteType::Http
278    };
279
280    if let Some(compiled_route) = routers.get_route(&req_method, router_type, &req_path) {
281        let client_origin = request.get_header(ORIGIN).map(|s| s.to_string());
282
283        let mut chain = compiled_route.filter_chain().clone();
284
285        let handler_arc = compiled_route.handler().clone();
286
287        let request_clone = request.clone();
288
289        let response_result = chain.execute::<_, ()>(request, move |req| {
290            let handler_arc_clone = handler_arc.clone();
291
292            let future: crate::utils::other::BoxFuture<'static, Result<Response<Bytes>, &'static str>> =
293                Box::pin(async move {
294                    match &handler_arc_clone {
295                        Handler::Http(_) => {
296                            Ok(handler_arc_clone.invoke_http(req).await)
297                        }
298                        Handler::WebSocket(_) => {
299                            Ok(Response::websocket_upgraded())
300                        }
301                        Handler::WebTransport(_) => {
302                            Ok(Response::default())
303                        }
304                    }
305                });
306
307            future
308        }).await;
309
310
311        let mut response = response_result.unwrap_or_else(|err_msg| {
312            println!("[ERROR] Ошибка выполнения цепочки фильтров: {}", err_msg);
313            Response::internal_error_body(err_msg)
314        });
315
316        if response.is_websocket_upgraded() {
317            if let Handler::WebSocket(ws_handler) = &*compiled_route.handler() {
318                if let Some(accept_key) = generate_accept_key(&request_clone) {
319                    let handshake_raw = format!(
320                        "HTTP/1.1 101 Switching Protocols\r\n\
321                 Upgrade: websocket\r\n\
322                 Connection: Upgrade\r\n\
323                 Sec-WebSocket-Accept: {}\r\n\r\n",
324                        accept_key
325                    );
326
327                    println!("[SHARK] Отправляем хэндшейк с ключом: {}", accept_key);
328
329                    stream
330                        .write_all(handshake_raw.as_bytes())
331                        .await
332                        .unwrap();
333                    stream.flush().await.unwrap();
334
335                    websocket_handler(stream, ws_handler, request_clone);
336                    return;
337                } else {
338                    response = Response::bad_request();
339                }
340            }
341        }
342
343        if let Some(origin_str) = client_origin
344            && !origin_str.is_empty()
345        {
346            response = response.header(ACCESS_CONTROL_ALLOW_ORIGIN, origin_str);
347        }
348
349        if let Err(e) = response.send(&mut stream).await {
350            info!("Error sending response: {}", e);
351        }
352    } else {
353        let client_origin = request.get_header(ORIGIN).unwrap_or("");
354        let mut response = Response::not_found();
355        if !client_origin.is_empty() {
356            response = response.header(ACCESS_CONTROL_ALLOW_ORIGIN, client_origin);
357        }
358        let _ = response.send(&mut stream).await;
359        stream.flush().await.unwrap();
360    }
361}

363fn websocket_handler<T>(stream: T, ws_handler: &WSHandler, request: Request<Bytes>)
364where
365    T: AsyncRead + AsyncWrite + Unpin + Sync + Send + 'static,
366{
367    let ctx = WebSocketContext::new(stream);
368
369    let future = ws_handler.invoke(request, ctx);
370
371    tokio::spawn(future);
372}
