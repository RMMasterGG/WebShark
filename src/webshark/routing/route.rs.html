1//! Модуль, описывающий структуру конкретного маршрута (Route).
2//!
3//! Объединяет в себе HTTP-метод, текстовый путь и стёртый по типу
4//! обработчик (Handler), превращая их в единую сущность для роутера.
5

15/// Структура, представляющая зарегистрированный эндпоинт.
16pub struct Route {
17    method: Method,
18    path: &'static str,
19    handler: Arc<dyn RouteHandler<Request<Bytes>> + Sync + Send + 'static>,

33impl Route {
34    /// Создает новый маршрут, автоматически затирая тип аргументов функции-обработчика.
35    ///
36    /// # Examples
37    ///
38    /// ```no_run
39    /// use webshark::routing::request::{Method, Request};
40    /// use webshark::routing::response::Response;
41    /// use webshark::routing::route::Route;
42    ///
43    /// // Функция без аргументов
44    /// fn index() -> Response { Response::ok() }
45    /// let route_home = Route::new(Method::GET, "/", index);
46    ///
47    /// // Функция с аргументом Request
48    /// fn api(req: Request) -> Response { Response::ok() }
49    /// let route_api = Route::new(Method::POST, "/api", api);
50    /// ```
51    pub fn new<H, Args>(method: Method, path: &'static str, handler: H) -> Self
52    where
53        H: RouteHandler<Args> + Sync + Send + 'static,
54        Args: Send + Sync + 'static,
55    {
56        let boxed = BoxedHandler {
57            inner: handler,
58            _marker: PhantomData,
59        };
60
61        Self {
62            method,
63            path,
64            handler: Arc::new(boxed),
65        }
66    }
67
68    /// Уничтожает структуру `Route` и возвращает владение упакованным обработчиком.
69    ///
70    /// Используется роутером при добавлении маршрута в `HashMap`.
71    pub fn handler(&self) -> Arc<dyn RouteHandler<Request<Bytes>> + Sync + Send + 'static> {
72        self.handler.clone()
73    }
74
75    /// Возвращает HTTP-метод маршрута.
76    pub fn method(&self) -> Method {
77        self.method.clone()
78    }
79
80    /// Возвращает текстовый путь маршрута.
81    pub fn path(&self) -> &'static str {
82        self.path
83    }
84}
