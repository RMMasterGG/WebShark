1//! Модуль, отвечающий за абстракцию и затирание типов функций-обработчиков (эндпоинтов).
2//!
3//! Позволяет роутеру регистрировать и вызывать как функции без аргументов,
4//! так и функции, принимающие объект [`Request`].
5
6use crate::Request;
7use crate::routing::socket_context::WebSocketContext;
8use bytes::Bytes;
9use crate::utils::other::{BoxFuture, BoxedHandler};
10
11/// Трейт, инкапсулирующий вызов эндпоинта.
12///
13/// Использует маркер обобщения `Args` для разделения функций с разной сигнатурой
14/// на этапе компиляции, предотвращая конфликт реализаций.
15pub trait SocketHandler<Args> {
16    fn invoke(&self, req: Request<Bytes>, ctx: WebSocketContext) -> BoxFuture<'static, ()>;
17}
18
19impl<F, Fut> SocketHandler<()> for F
20where
21    F: Fn() -> Fut + Send + Sync + 'static,
22    Fut: Future<Output=()> + Send + 'static,
23{
24    fn invoke(&self, _req: Request<Bytes>, _ctx: WebSocketContext) -> BoxFuture<'static, ()> {
25        let fut = self();
26        Box::pin(async move { fut.await })
27    }
28}
29
30impl<F, Fut> SocketHandler<Request<Bytes>> for F
31where
32    F: Fn(Request<Bytes>) -> Fut + Send + Sync + 'static,
33    Fut: Future<Output=()> + Send + 'static,
34{
35    fn invoke(&self, req: Request<Bytes>, _ctx: WebSocketContext) -> BoxFuture<'static, ()> {
36        let fut = self(req);
37        Box::pin(async move { fut.await })
38    }
39}
40
41impl<F, Fut> SocketHandler<WebSocketContext> for F
42where
43    F: Fn(WebSocketContext) -> Fut + Send + Sync + 'static,
44    Fut: Future<Output=()> + Send + 'static,
45{
46    fn invoke(&self, _req: Request<Bytes>, ctx: WebSocketContext) -> BoxFuture<'static, ()> {
47        let fut = self(ctx);
48        Box::pin(async move { fut.await })
49    }
50}
51
52impl<H, Args> SocketHandler<(Request<Bytes>, WebSocketContext)> for BoxedHandler<H, Args>
53where
54    H: SocketHandler<Args> + Sync + Send + 'static,
55    Args: Send + Sync + 'static,
56{
57    fn invoke(
58        &self,
59        req: Request<Bytes>,
60        ctx: WebSocketContext,
61    ) -> BoxFuture<'static, ()> {
62        self.inner.invoke(req, ctx)
63    }
64}
65
66/// Для функций fn(req, conn)
67impl<F, Fut> SocketHandler<(Request<Bytes>, WebSocketContext)> for F
68where
69    F: Fn(Request<Bytes>, WebSocketContext) -> Fut + Send + Sync + 'static,
70    Fut: Future<Output=()> + Send + 'static,
71{
72    fn invoke(&self, req: Request<Bytes>, ctx: WebSocketContext) -> BoxFuture<'static, ()> {
73        let fut = self(req, ctx);
74        Box::pin(async move { fut.await })
75    }
76}
