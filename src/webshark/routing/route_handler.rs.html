1//! Модуль, отвечающий за абстракцию и затирание типов функций-обработчиков (эндпоинтов).
2//!
3//! Позволяет роутеру регистрировать и вызывать как функции без аргументов,
4//! так и функции, принимающие объект [`Request`].
5
6use bytes::Bytes;
7use crate::{Request, Response};
8use crate::utils::other::{BoxFuture, BoxedHandler};
9
10
11/// Трейт, инкапсулирующий вызов эндпоинта.
12///
13/// Использует маркер обобщения `Args` для разделения функций с разной сигнатурой
14/// на этапе компиляции, предотвращая конфликт реализаций.
15pub trait RouteHandler<Args> {
16    fn invoke(&self, req: Request<Bytes>) -> BoxFuture<'static, Response<Bytes>>;
17}
18
19/// Реализация для функций и замыканий, не принимающих аргументов.
20impl<F, Fut> RouteHandler<()> for F
21where
22    F: Fn() -> Fut + Send + Sync + 'static,
23    Fut: Future<Output = Response<Bytes>> + Send + 'static,
24{
25    fn invoke(&self, _req: Request<Bytes>) -> BoxFuture<'static, Response<Bytes>> {
26        let fut = self();
27        Box::pin(fut)
28    }
29}
30
31/// Реализация для функций и замыканий, принимающих [`Request`] по значению.
32impl<F, Fut> RouteHandler<Request<Bytes>> for F
33where
34    F: Fn(Request<Bytes>) -> Fut + Send + Sync + 'static,
35    Fut: Future<Output = Response<Bytes>> + Send + 'static,
36{
37    fn invoke(&self, req: Request<Bytes>) -> BoxFuture<'static, Response<Bytes>> {
38        let fut = self(req);
39        Box::pin(fut)
40    }
41}
42
43impl<H, Args> RouteHandler<(Request<Bytes>)> for BoxedHandler<H, Args>
44where
45    H: RouteHandler<Args> + Sync + Send + 'static,
46    Args: Send + Sync + 'static,
47{
48    fn invoke(
49        &self,
50        req: Request<Bytes>
51    ) -> BoxFuture<'static, Response<Bytes>> {
52        self.inner.invoke(req)
53    }
54}
