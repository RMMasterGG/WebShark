24/// Структура HTTP-ответа.
25///
26/// Использует паттерн «Строитель» (Builder) для поэтапной настройки
27/// статуса, типа контента и тела ответа перед отправкой.
28#[derive(Debug, Default)]
29pub struct Response<B> {
30    inner: HttpResponse<B>,
31}

33impl Display for Response<Bytes> {
34    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
35        let raw_body = String::from_utf8_lossy(&self.inner.body());
36
37        if f.alternate() {
38            write!(
39                f,
40                "\nStatus: {},\nBody:\n{}",
41                self.inner.status().as_u16(),
42                raw_body.trim()
43            )
44        } else {
45            let clean_body = raw_body.replace("\n", " ").replace("\r", "");
46            write!(
47                f,
48                "Status: {}, Body: {}",
49                self.inner.status().as_u16(),
50                clean_body.trim()
51            )
52        }
53    }
54}

56impl Response<Bytes> {
57    pub fn to_bytes(&self) -> Vec<u8> {
58        let mut buf = Vec::with_capacity(512);
59
60        let status = self.inner.status();
61
62        let reason = status.canonical_reason().unwrap_or("");
63        let status_u16 = status.as_u16().to_string();
64
65        let mut status_line = String::with_capacity(12 + status_u16.len() + reason.len());
66        status_line.push_str("HTTP/1.1 ");
67        status_line.push_str(&status_u16);
68        status_line.push_str(" ");
69        status_line.push_str(reason);
70        status_line.push_str("\r\n");
71
72        buf.extend_from_slice(status_line.as_bytes());
73
74        for (name, value) in self.inner.headers() {
75            buf.extend_from_slice(name.as_str().as_bytes());
76            buf.extend_from_slice(b": ");
77            buf.extend_from_slice(value.as_bytes());
78            buf.extend_from_slice(b"\r\n");
79        }
80        buf.extend_from_slice(b"\r\n");
81
82        if !self.inner.body().is_empty() {
83            buf.extend_from_slice(self.inner.body());
84        }
85
86        buf
87    }
88}

91    /// Создает базовый пустой ответ со статусом `200 OK` и типом `text/html`.
92    pub fn new() -> Self {

96    /// Возвращает текущий HTTP-статус ответа.
97    /// (HttpStatus копируемый, поэтому возвращаем по значению)
98    pub fn get_status(&self) -> StatusCode {
99        self.inner.status()
100    }

102    pub fn is_websocket_upgraded(&self) -> bool {
103        self.get_status() == StatusCode::SWITCHING_PROTOCOLS
104    }
105
106    pub fn get_content_type(&self) -> Option<&str> {
107        self.inner
108            .headers()
109            .get(CONTENT_TYPE)
110            .and_then(|v| v.to_str().ok())
111    }
112
113    /// Возвращает ссылку на бинарное тело ответа.
114    pub fn get_body(&self) -> &[u8] {
115        self.inner.body().as_ref()
116    }
117
118    /// Пытается интерпретировать тело ответа как UTF-8 строку.
119    /// Полезно для логирования или отладки.
120    pub fn body_as_str(&self) -> std::borrow::Cow<'_, str> {
121        String::from_utf8_lossy(&self.inner.body())
122    }
123
124    /// Устанавливает HTTP-статус ответа.
125    pub fn status(mut self, status: StatusCode) -> Self {
126        *self.inner.status_mut() = status;
127        self
128    }
129
130    pub fn html(self) -> Self {
131        self.content_type(TEXT_HTML_UTF_8)
132    }
133
134    pub fn json(self) -> Self {
135        self.content_type(APPLICATION_JSON)
136    }
137
138    pub fn content_type(mut self, value: impl AsRef<str>) -> Self {
139        let value_str = value.as_ref();
140        if let Ok(content_type) = value_str.parse::<Mime>() {
141            if let Ok(header_value) = HeaderValue::from_str(content_type.as_ref()) {
142                self.inner.headers_mut().insert(CONTENT_TYPE, header_value);
143            }
144        } else {
145            warn!("Не удалось распознать тип контента");
146        }
147        self
148    }
149
150    /// Устанавливаем кастомные заголовки.
151    pub fn header<T>(mut self, key: T, value: impl AsRef<str>) -> Self
152    where
153        T: TryInto<HeaderName>,
154        T::Error: Debug,
155    {
156        let val_str = value.as_ref();
157
158        let result = key
159            .try_into()
160            .map_err(|_| HeaderError::InvalidName)
161            .and_then(|header_name| {
162                HeaderValue::try_from(val_str)
163                    .map_err(HeaderError::InvalidValue)
164                    .map(|header_value| (header_name, header_value))
165            });
166
167        match result {
168            Ok((header_name, header_value)) => {
169                self.inner.headers_mut().insert(header_name, header_value);
170            }
171            Err(HeaderError::InvalidName) => {
172                warn!("Некорректное имя заголовка");
173            }
174            Err(HeaderError::InvalidValue(err)) => {
175                warn!(
176                    "Некорректные символы в значении заголовка: {}. Ошибка: {:?}",
177                    val_str, err
178                );
179            }
180        }
181        self
182    }
183
184    /// Записывает данные в тело ответа.
185    ///
186    /// Принимает любые типы, конвертируемые в `Bytes` (например, `&str`, `String`, `Vec<u8>`, `&[u8]`).
187    pub fn body(mut self, body: impl Into<Bytes>) -> Self {
188        *self.inner.body_mut() = body.into();
189        self
190    }

192    /// Добавляет заголовок `Set-Cookie`.
193    pub fn set_cookie(self, cookie: Cookie) -> Self {
194        self.header(SET_COOKIE, cookie.to_string())
195    }

197    /// Сериализует ответ в формате HTTP/1.1, отправляет его в поток и возвращает `Self` для логов.
198    ///
199    /// Метод полностью забирает владение структурой (`self`). Работает с любым объектом,
200    /// реализующим трейт [`Write`].
201    pub async fn send(self, stream: &mut (impl AsyncWrite + Unpin)) -> Result<Self, Error> {
202        let mut header_buf = Vec::with_capacity(512);
203
204        let status = self.inner.status();
205        let status_phrase = status.canonical_reason().unwrap_or("");
206
207        header_buf.extend(
208            format_args!("HTTP/1.1 {} {}\r\n", status.as_u16(), status_phrase)
209                .to_string()
210                .as_bytes(),
211        );
212
213        let has_body = !status.is_informational()
214            && status != StatusCode::NO_CONTENT
215            && status != StatusCode::NOT_MODIFIED;
216
217        if has_body {
218            header_buf.extend(
219                format_args!("content-length: {}\r\n", self.inner.body().len())
220                    .to_string()
221                    .as_bytes(),
222            );
223        }
224
225        // Headers
226        for (name, value) in self.inner.headers().iter() {
227            header_buf.extend_from_slice(name.as_str().as_bytes());
228            header_buf.extend_from_slice(b": ");
229            header_buf.extend_from_slice(value.as_bytes());
230            header_buf.extend_from_slice(b"\r\n");
231        }
232
233        if !self.inner.headers().contains_key(CONNECTION) {
234            header_buf.extend_from_slice(b"connection: close\r\n");
235        }
236
237        header_buf.write_all(b"\r\n").await?;
238        stream.write_all(&header_buf).await?;
239
240        if has_body && !self.inner.body().is_empty() {
241            stream.write_all(self.inner.body()).await?;
242        }
243
244        stream.flush().await?;
245        Ok(self)
246    }
247}

249/// Ассоциированные статические методы для быстрого формирования ответов в стиле Spring.
250///
251/// Позволяют генерировать готовые объекты ответов без вызова конструктора [`Response::new`].
252///
253/// # Examples
254///
255/// ```no_run
256/// use webshark::routing::response::Response;
257///
258/// // Быстрый пустой успешный ответ
259/// let res_ok = Response::ok();
260///
261/// // Создание ресурса с JSON-телом
262/// let json_data = r#"{"id": 42}"#;
263/// let res_created = Response::created_body(json_data).content_type("application/json");
264///
265/// // Ошибка 404 с кастомным сообщением
266/// let res_not_found = Response::not_found_body("Страница не существует");
267/// ```
268impl Response<Bytes> {
269    /// Быстрый пустой ответ `200 OK`.
270    pub fn ok() -> Self {
271        Self::new().status(StatusCode::OK)
272    }
273
274    /// Быстрый ответ `200 OK` со строковым или бинарным телом.
275    ///
276    /// Принимает любые типы, конвертируемые в `Vec<u8>`.
277    ///
278    /// # Examples
279    ///
280    /// ```no_run
281    /// use webshark::routing::response::Response;
282    ///
283    /// let res_str = Response::ok_body("Привет, мир!");
284    /// let res_bytes = Response::ok_body(b"Hello".to_vec());
285    /// ```
286    pub fn ok_body(body: impl Into<Bytes>) -> Self {
287        Self::new().status(StatusCode::OK).body(body)
288    }
289
290    /// Быстрый пустой ответ `201 Created`.
291    pub fn created() -> Self {
292        Self::new().status(StatusCode::CREATED)
293    }
294
295    /// Быстрый ответ `201 Created` со строковым или бинарным телом ресурса.
296    ///
297    /// Принимает любые типы, конвертируемые в `Vec<u8>`.
298    ///
299    /// # Examples
300    ///
301    /// ```no_run
302    /// use webshark::routing::response::Response;
303    ///
304    /// let res = Response::created_body("User registered successfully");
305    /// ```
306    pub fn created_body(body: impl Into<Bytes>) -> Self {
307        Self::new().status(StatusCode::CREATED).body(body)
308    }
309
310    /// Быстрый пустой ответ `204 No Content`.
311    ///
312    /// Гарантирует строгое отсутствие заголовков длины, типа и самого тела при отправке.
313    pub fn no_content() -> Self {
314        Self::new().status(StatusCode::NO_CONTENT)
315    }
316
317    /// Быстрый пустой ответ `400 Bad Request`.
318    pub fn bad_request() -> Self {
319        Self::new().status(StatusCode::BAD_REQUEST)
320    }
321
322    pub fn websocket_upgraded() -> Self {
323        Self::new().status(StatusCode::SWITCHING_PROTOCOLS)
324    }
325
326    pub fn websocket_full_upgraded(accept_key: impl Into<String>) -> Self {
327        Self::new()
328            .status(StatusCode::SWITCHING_PROTOCOLS)
329            .header(UPGRADE, "websocket")
330            .header(CONNECTION, "upgrade")
331            .header(SEC_WEBSOCKET_ACCEPT, accept_key.into())
332    }
333
334    /// Быстрый пустой ответ `404 Not Found`.
335    pub fn not_found() -> Self {
336        Self::new().status(StatusCode::NOT_FOUND)
337    }
338
339    pub fn internal_error() -> Self {
340        Self::new().status(StatusCode::INTERNAL_SERVER_ERROR)
341    }
342
343    pub fn internal_error_body(body: impl Into<Bytes>) -> Self {
344        Self::new()
345            .status(StatusCode::INTERNAL_SERVER_ERROR)
346            .body(body)
347    }
348
349    /// Быстрый ответ `404 Not Found` с текстовым описанием ошибки.
350    ///
351    /// Принимает любые типы, конвертируемые в `Vec<u8>`.
352    ///
353    /// # Examples
354    ///
355    /// ```no_run
356    /// use webshark::routing::response::Response;
357    ///
358    /// let res = Response::not_found_body("Файл index.html не найден на диске");
359    /// ```
360    pub fn not_found_body(message: impl Into<Vec<u8>>) -> Self {
361        Self::new()
362            .status(StatusCode::NOT_FOUND)
363            .body(message.into())
364    }
365
366    pub fn forbidden() -> Self {
367        Self::new().status(StatusCode::FORBIDDEN)
368    }
369}
